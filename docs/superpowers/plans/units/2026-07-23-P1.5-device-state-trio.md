## Goal

Move the three device-state helper modules that still live under
`apps/api/src/services/` into `@www/core`, so the ac / ctrl / sound enforcer
cluster can later fold into `features/*` without crossing the Biome-banned
`features/* -> apps/api/*` boundary. Three sub-moves, one commit:

- **(a) command-window** — `apps/api/src/services/command-window.ts` is already a
  pure re-export of `@www/core`. Delete the shim, repoint consumers to `@www/core`.
- **(b) device-state-mapping** — split the HA-shape mapping + value-comparison
  functions (pure, `HaEntity` already lives in core) into a new core module; keep
  the config-coupled ownership routing (`ownerOf`, which needs `findLight` from
  `apps/api/src/config/lights.ts`) in `apps/api`.
- **(c) integration-heartbeat** — needs `db` + the `integration_sync_status`
  table. Per the master-plan resolution, move the table into `@www/core` behind a
  new `IntegrationSyncStore` interface with pg + memory adapters, mirroring the
  `device_state` precedent. `heartbeat`/`runCycle` become store-driven helpers in
  core.

Not in this unit: `sonos-volume-enforcer -> features/sound` (a later FOLD). The
enforcer services stay in `apps/api` here; they only get their imports repointed.

**`store.ts`** — the interface (the ONLY code that touches the table; DB failures
always throw, mirroring `DeviceStateStore`'s doc contract):

```ts
import type { integrationSyncStatus } from "./schema";

export type IntegrationSyncRow = typeof integrationSyncStatus.$inferSelect;

export interface IntegrationSyncStore {
  /** Read the row for an integration, or null when it has never reported. */
  read(integrationId: string): Promise<IntegrationSyncRow | null>;
  /** Record a successful cycle: fresh poll time, no error, streak reset to 0. */
  recordOk(integrationId: string): Promise<void>;
  /** Record a failed cycle; returns the NEW consecutive-failure streak. */
  recordFail(integrationId: string, error: string): Promise<number>;
}
```

**`pg.ts`** — pg adapter (mirror `device-state/pg.ts`'s `Pick<NodePgDatabase, ...>`
surface; issues the SAME drizzle calls the old `heartbeat` did, so apps/api tests
that mock `db` keep capturing identical ops):

```ts
import { eq } from "drizzle-orm";
import type { NodePgDatabase } from "drizzle-orm/node-postgres";
import { integrationSyncStatus } from "./schema";
import type { IntegrationSyncRow, IntegrationSyncStore } from "./store";

export type PgIntegrationSyncDb = Pick<
  NodePgDatabase<Record<string, unknown>>,
  "select" | "insert"
>;

export function createPgIntegrationSyncStore(db: PgIntegrationSyncDb): IntegrationSyncStore {
  async function write(integrationId: string, error: string | null, consecutiveFailures: number) {
    const now = new Date();
    await db
      .insert(integrationSyncStatus)
      .values({ integrationId, lastPolledAtUtc: now, lastError: error, consecutiveFailures })
      .onConflictDoUpdate({
        target: integrationSyncStatus.integrationId,
        set: { lastPolledAtUtc: now, lastError: error, consecutiveFailures },
      });
  }

  return {
    async read(integrationId) {
      const rows = await db
        .select()
        .from(integrationSyncStatus)
        .where(eq(integrationSyncStatus.integrationId, integrationId))
        .limit(1);
      return rows[0] ?? null;
    },
    async recordOk(integrationId) {
      await write(integrationId, null, 0);
    },
    async recordFail(integrationId, error) {
      const rows = await db
        .select({ n: integrationSyncStatus.consecutiveFailures })
        .from(integrationSyncStatus)
        .where(eq(integrationSyncStatus.integrationId, integrationId))
        .limit(1);
      const consecutiveFailures = (rows[0]?.n ?? 0) + 1;
      await write(integrationId, error, consecutiveFailures);
      return consecutiveFailures;
    },
  };
}
```

**`memory.ts`** — in-memory adapter (`Map<integrationId, IntegrationSyncRow>`),
mirroring `device-state/memory.ts` (structuredClone on read).

> NOTE (intentional, not a bug): the pg `write` faithfully MIRRORS the original
> `heartbeat` and OMITS `updatedAtUtc` from `onConflictDoUpdate.set`, so on update
> the column keeps its first-insert `defaultNow()` value — required for "no
> behavior change". The memory adapter below DOES set `updatedAtUtc: now` on every
> write; this is a harmless divergence (memory is test-only, no assertion reads the
> column). Do NOT "harmonize" by adding `updatedAtUtc` to the pg conflict-set.

```ts
import type { IntegrationSyncRow, IntegrationSyncStore } from "./store";

export function createInMemoryIntegrationSyncStore(): IntegrationSyncStore {
  const rows = new Map<string, IntegrationSyncRow>();
  function write(integrationId: string, error: string | null, consecutiveFailures: number) {
    const now = new Date();
    rows.set(integrationId, {
      integrationId, lastPolledAtUtc: now, lastError: error, consecutiveFailures, updatedAtUtc: now,
    });
  }
  return {
    async read(integrationId) {
      const row = rows.get(integrationId);
      return row ? structuredClone(row) : null;
    },
    async recordOk(integrationId) { write(integrationId, null, 0); },
    async recordFail(integrationId, error) {
      const consecutiveFailures = (rows.get(integrationId)?.consecutiveFailures ?? 0) + 1;
      write(integrationId, error, consecutiveFailures);
      return consecutiveFailures;
    },
  };
}
```

**`heartbeat.ts`** — the store-driven helpers (move `IntegrationHeartbeat` +
`runCycle` bodies verbatim; `heartbeat` now takes a store):

```ts
import { getLogger } from "@www/logger";
import type { IntegrationSyncStore } from "./store";

export interface IntegrationHeartbeat {
  ok(): Promise<void>;
  fail(error: string): Promise<number>;
}

export function heartbeat(store: IntegrationSyncStore, integrationId: string): IntegrationHeartbeat {
  return {
    async ok() { await store.recordOk(integrationId); },
    async fail(error: string) { return store.recordFail(integrationId, error); },
  };
}

export async function runCycle(
  hb: IntegrationHeartbeat,
  label: string,
  work: () => Promise<void>,
): Promise<void> {
  try {
    await work();
    await hb.ok();
  } catch (err) {
    const consecutiveFailures = await hb.fail(err instanceof Error ? err.message : String(err));
    getLogger().error({ err, consecutiveFailures }, `${label} cycle failed`);
  }
}
```
